#### configMap

```yaml
apiVersion: v1
kind: ConfigMap
metadata:
  name: log-config
data:
  log_level: debug
---

```

```shell
cat <<EOF | sudo tee configmap-pod.yaml
apiVersion: v1
kind: Pod
metadata:
  name: configmap-pod
spec:
  containers:
    - name: test
      image: nginx
      volumeMounts:
        - name: config-vol
          mountPath: /etc/config
  volumes:
    - name: config-vol
      configMap:
        name: log-config
        items:
          - key: log_level
            path: log_level
EOF
```
